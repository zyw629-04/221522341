def generate_sample_data(n_samples=840, n_dimensions=7, n_items_per_dim=3, random_seed=42):
    """生成模拟问卷调查数据
    
    参数:
        n_samples: 样本数量
        n_dimensions: 维度数量
        n_items_per_dim: 每个维度的题目数量
        random_seed: 随机种子
    """
    np.random.seed(random_seed)
    
    # 创建模拟数据
    data = {}
    all_items = []
    
    for dim_idx in range(n_dimensions):
        dimension_name = f"维度{dim_idx+1}"
        for item_idx in range(n_items_per_dim):
            item_name = f"Q{dim_idx*n_items_per_dim + item_idx + 1}"
            data[item_name] = np.random.normal(3.5, 0.8, n_samples)
            all_items.append(item_name)
    
    # 创建维度-题目映射
    dimensions = {}
    for dim_idx in range(n_dimensions):
        dimension_name = f"维度{dim_idx+1}"
        items = [f"Q{dim_idx*n_items_per_dim + item_idx + 1}" for item_idx in range(n_items_per_dim)]
        dimensions[dimension_name] = items
    
    return pd.DataFrame(data), dimensions, all_items

def cronbach_alpha(data):
    """计算Cronbach's α信度系数"""
    n_items = data.shape[1]
    item_var = data.var(axis=0, ddof=1)
    total_var = data.sum(axis=1).var(ddof=1)
    
    alpha = (n_items / (n_items - 1)) * (1 - item_var.sum() / total_var)
    return alpha

def reliability_analysis(df, dimensions):
    """执行信度分析
    
    参数:
        df: 数据DataFrame
        dimensions: 维度-题目映射字典
    """
    results = {}
    
    for dimension, items in dimensions.items():
        valid_items = [item for item in items if item in df.columns]
        
        if len(valid_items) >= 2:
            alpha = cronbach_alpha(df[valid_items].values)
            results[dimension] = {
                "alpha": alpha,
                "items": valid_items,
                "status": "良好" if alpha >= 0.7 else "不足"
            }
        else:
            results[dimension] = {
                "alpha": None,
                "message": f"题目数量不足: 需要至少2题，实际{len(valid_items)}题"
            }
    
    return results

def validity_analysis(df, items, n_factors=None):
    """执行探索性因子分析(EFA)
    
    参数:
        df: 数据DataFrame
        items: 参与分析的题目列表
        n_factors: 提取的因子数量(默认None，表示自动确定)
    """
    # 筛选有效题目
    valid_items = [item for item in items if item in df.columns]
    
    if len(valid_items) < 3:
        return {"error": f"题目数量不足: 需要至少3题，实际{len(valid_items)}题"}
    
    # 数据标准化
    scaler = StandardScaler()
    df_std = scaler.fit_transform(df[valid_items])
    
    # Bartlett球形检验
    chi_square, p_value = calculate_bartlett_sphericity(df_std)
    
    # KMO检验
    kmo_all, kmo_model = calculate_kmo(df_std)
    
    # 计算特征值以确定因子数量
    fa_pre = FactorAnalyzer(n_factors=len(valid_items), rotation=None)
    fa_pre.fit(df_std)
    ev, v = fa_pre.get_eigenvalues()
    
    # 如果未指定因子数量，使用特征值>1的规则
    if n_factors is None:
        n_factors = sum(ev > 1)
    
    # 执行因子分析
    fa = FactorAnalyzer(n_factors=n_factors, rotation='varimax')
    fa.fit(df_std)
    
    # 获取结果
    loadings = fa.loadings_
    variance = fa.get_factor_variance()
    
    return {
        "kmo": kmo_model,
        "bartlett": {"chi_square": chi_square, "p_value": p_value},
        "n_factors": n_factors,
        "eigenvalues": ev,
        "loadings": loadings,
        "variance": variance,
        "items": valid_items
    }

def plot_reliability_results(reliability_results):
    """绘制信度分析结果"""
    plt.figure(figsize=(10, 6))
    
    alpha_values = [result["alpha"] for result in reliability_results.values() if result.get("alpha") is not None]
    dimension_names = [dim for dim, result in reliability_results.items() if result.get("alpha") is not None]
    
    sns.barplot(x=dimension_names, y=alpha_values)
    plt.title('各维度Cronbach\'s α系数')
    plt.ylabel('α系数')
    plt.axhline(y=0.7, color='r', linestyle='--', label='可接受阈值')
    plt.xticks(rotation=45)
    plt.legend()
    
    plt.tight_layout()
    return plt

def plot_validity_results(validity_results):
    """绘制效度分析结果"""
    plt.figure(figsize=(16, 12))
    
    # 图1：碎石图
    plt.subplot(2, 2, 1)
    ev = validity_results['eigenvalues']
    plt.plot(range(1, len(ev)+1), ev, 'o-')
    plt.title('碎石图 (Scree Plot)')
    plt.xlabel('因子序号')
    plt.ylabel('特征值')
    plt.axhline(y=1, color='r', linestyle='--')
    plt.grid(True)
    
    # 图2：因子载荷热图
    plt.subplot(2, 2, 2)
    loadings_df = pd.DataFrame(
        validity_results['loadings'],
        columns=[f'因子{i+1}' for i in range(validity_results['n_factors'])],
        index=validity_results['items']
    )
    sns.heatmap(loadings_df, annot=True, cmap='coolwarm', fmt='.2f')
    plt.title('因子载荷矩阵热图')
    
    # 图3：方差解释率
    plt.subplot(2, 2, 3)
    variance = validity_results['variance']
    plt.bar(range(1, len(variance[1])+1), variance[1], label='方差贡献率')
    plt.plot(range(1, len(variance[2])+1), variance[2], 'o-', color='r', label='累积方差贡献率')
    plt.title('因子方差解释率')
    plt.xlabel('因子序号')
    plt.ylabel('方差贡献率')
    plt.legend()
    
    plt.tight_layout()
    return plt

def main():
    """主函数：执行完整的信效度分析流程"""
    print("开始信效度分析...")
    
    # 1. 生成模拟数据
    print("生成模拟数据...")
    df, dimensions, all_items = generate_sample_data(
        n_samples=840, 
        n_dimensions=7, 
        n_items_per_dim=3
    )
    print(f"数据生成成功：{df.shape[0]}个样本，{df.shape[1]}个变量")
    
    # 2. 信度分析
    print("执行信度分析...")
    reliability_results = reliability_analysis(df, dimensions)
    
    print("\n【信度分析结果】")
    for dimension, result in reliability_results.items():
        if result.get("alpha") is not None:
            print(f"{dimension}: α = {result['alpha']:.4f} ({result['status']})")
        else:
            print(f"{dimension}: {result['message']}")
    
    # 3. 效度分析
    print("\n执行效度分析...")
    validity_results = validity_analysis(df, all_items, n_factors=7)
    
    if "error" in validity_results:
        print(validity_results["error"])
        return
    
    print("\n【效度分析结果】")
    print(f"KMO检验: {validity_results['kmo']:.4f} ({'适合因子分析' if validity_results['kmo'] >= 0.6 else '不适合'})")
    print(f"Bartlett球形检验: χ²={validity_results['bartlett']['chi_square']:.2f}, p={validity_results['bartlett']['p_value']:.4e}")
    print(f"提取因子数量: {validity_results['n_factors']}")
    
    print("\n【方差解释率】")
    variance = validity_results['variance']
    for i in range(validity_results['n_factors']):
        print(f"因子 {i+1}: 特征值={variance[0][i]:.2f}, 方差贡献率={variance[1][i]:.2%}, 累积方差贡献率={variance[2][i]:.2%}")
    
    print("\n【因子载荷矩阵】")
    loadings_df = pd.DataFrame(
        validity_results['loadings'],
        columns=[f'因子{i+1}' for i in range(validity_results['n_factors'])],
        index=validity_results['items']
    )
    print(loadings_df.round(4))
    
    # 4. 可视化结果
    print("\n生成可视化图表...")
    
    # 信度分析图表
    rel_plt = plot_reliability_results(reliability_results)
    rel_plt.savefig('信度分析结果.png', dpi=300)
    
    # 效度分析图表
    val_plt = plot_validity_results(validity_results)
    val_plt.savefig('效度分析结果.png', dpi=300)
    
    print("\n分析完成！")
    print("信度分析结果已保存到'信度分析结果.png'")
    print("效度分析结果已保存到'效度分析结果.png'")

if __name__ == "__main__":
    main()
