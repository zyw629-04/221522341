import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import StandardScaler
import statsmodels.api as sm
from statsmodels.stats.outliers_influence import variance_inflation_factor
from statsmodels.stats.multicomp import pairwise_tukeyhsd
from scipy.stats import f_oneway

# 1. 
np.random.seed(42)
n = 840  # 840
df = pd.DataFrame({
    # 13.1-13.3
    '13.1': np.random.normal(3.5, 1, n),
    '13.2': np.random.normal(3.5, 1, n),
    '13.3': np.random.normal(3.5, 1, n),
    # 14.1-14.3
    '14.1': np.random.normal(3.5, 1, n),
    '14.2': np.random.normal(3.5, 1, n),
    '14.3': np.random.normal(3.5, 1, n),
    # 15.1-15.3
    '15.1': np.random.normal(3.5, 1, n),
    '15.2': np.random.normal(3.5, 1, n),  # 
    '15.3': np.random.normal(3.5, 1, n),
    # 16.1-16.3
    '16.1': np.random.normal(3.5, 1, n),
    '16.2': np.random.normal(3.5, 1, n),
    '16.3': np.random.normal(3.5, 1, n),
    # 20.1-20.3
    '20.1': np.random.normal(3.5, 1, n),
    '20.2': np.random.normal(3.5, 1, n),
    '20.3': np.random.normal(3.5, 1, n)
})
print("{}".format(df.shape[0]))

# 2. TAM
def calculate_variable_score(df, cols):
    return df[cols].mean(axis=1)

variable_mapping = {
    'website_quality': ['13.1', '13.2', '13.3'],    # 
    'perceived_ease': ['14.1', '14.2', '14.3'],     # 
    'perceived_utility': ['15.1', '15.3'],          # 
    'convenience': ['16.1', '16.2', '16.3'],       # 
    'shopping_behavior': ['20.1', '20.2', '20.3'], # 
    'info_completeness': ['15.2']                   # 
}

for var, cols in variable_mapping.items():
    df[var] = calculate_variable_score(df, cols)
    print(f" {var} {cols}")

# 3. 
scaler = StandardScaler()
selected_vars = ['website_quality', 'perceived_ease', 'perceived_utility', 'convenience']
X_scaled = scaler.fit_transform(df[selected_vars])
X_scaled = pd.DataFrame(X_scaled, columns=selected_vars)
y = df['shopping_behavior']

# 4. H1-H5
model = LinearRegression()
model.fit(X_scaled, y)
print("\n")
for i, var in enumerate(selected_vars):
    print(f"{var} : {model.coef_[i]:.4f}")

# statsmodels
X_with_const = sm.add_constant(X_scaled)
results = sm.OLS(y, X_with_const).fit()
print("\n")
print(results.summary().tables[1])  # 

# 5. VIF
vif = pd.DataFrame()
vif[""] = X_scaled.columns
vif["VIF"] = [variance_inflation_factor(X_scaled.values, i) 
              for i in range(X_scaled.shape[1])]
print("\n(VIF)")
print(vif)

# 6. 
df['group'] = pd.cut(df['info_completeness'], bins=5, labels=[1, 2, 3, 4, 5])
target_vars = ['website_quality', 'perceived_ease', 'perceived_utility', 'shopping_behavior']

print("\n")
for var in target_vars:
    groups = [df[df['group'] == i][var].dropna() for i in range(1, 6) 
              if not df[df['group'] == i][var].empty]
    if len(groups) >= 2:
        f_stat, p_value = f_oneway(*groups)
        print(f"\n{var} :")
        print(f"F: {f_stat:.4f}, p: {p_value:.4f}, : {'' if p_value < 0.05 else ''}")
        
        # Tukey HSD
        df_tukey = pd.DataFrame({'value': df[var], 'group': df['group'].astype(str)})
        tukey = pairwise_tukeyhsd(endog=df_tukey['value'], groups=df_tukey['group'], alpha=0.05)
        print("Tukey HSD:")
        tukey_df = pd.DataFrame(tukey._results_table.data[1:], columns=tukey._results_table.data[0])
        print(tukey_df.head(3))  # 3

# 7. 
plt.figure(figsize=(18, 12))
