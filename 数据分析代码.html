import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.linear_model import LinearRegression
from sklearn.preprocessing import StandardScaler
import statsmodels.api as sm
from statsmodels.stats.outliers_influence import variance_inflation_factor
from statsmodels.stats.multicomp import pairwise_tukeyhsd
from scipy.stats import f_oneway

# 1. 生成模拟数据（匹配文档中的变量定义）
np.random.seed(42)
n = 840  # 模拟840份有效问卷
df = pd.DataFrame({
    # 网站质量（问题13.1-13.3）
    '13.1': np.random.normal(3.5, 1, n),
    '13.2': np.random.normal(3.5, 1, n),
    '13.3': np.random.normal(3.5, 1, n),
    # 感知易用性（问题14.1-14.3）
    '14.1': np.random.normal(3.5, 1, n),
    '14.2': np.random.normal(3.5, 1, n),
    '14.3': np.random.normal(3.5, 1, n),
    # 感知有用性（问题15.1-15.3）
    '15.1': np.random.normal(3.5, 1, n),
    '15.2': np.random.normal(3.5, 1, n),  # 信息全面性（用于分组）
    '15.3': np.random.normal(3.5, 1, n),
    # 便利性（问题16.1-16.3）
    '16.1': np.random.normal(3.5, 1, n),
    '16.2': np.random.normal(3.5, 1, n),
    '16.3': np.random.normal(3.5, 1, n),
    # 实际购物行为（问题20.1-20.3）
    '20.1': np.random.normal(3.5, 1, n),
    '20.2': np.random.normal(3.5, 1, n),
    '20.3': np.random.normal(3.5, 1, n)
})
print("模拟数据生成成功，共{}条记录".format(df.shape[0]))

# 2. 生成研究变量（对应文档中的TAM模型变量）
def calculate_variable_score(df, cols):
    return df[cols].mean(axis=1)

variable_mapping = {
    'website_quality': ['13.1', '13.2', '13.3'],    # 网站质量
    'perceived_ease': ['14.1', '14.2', '14.3'],     # 感知易用性
    'perceived_utility': ['15.1', '15.3'],          # 感知有用性
    'convenience': ['16.1', '16.2', '16.3'],       # 便利性
    'shopping_behavior': ['20.1', '20.2', '20.3'], # 实际购物行为
    'info_completeness': ['15.2']                   # 信息全面性（分组变量）
}

for var, cols in variable_mapping.items():
    df[var] = calculate_variable_score(df, cols)
    print(f"成功生成变量 {var}，基于问题 {cols}")

# 3. 数据标准化（用于回归分析）
scaler = StandardScaler()
selected_vars = ['website_quality', 'perceived_ease', 'perceived_utility', 'convenience']
X_scaled = scaler.fit_transform(df[selected_vars])
X_scaled = pd.DataFrame(X_scaled, columns=selected_vars)
y = df['shopping_behavior']

# 4. 线性回归分析（验证文档中的H1-H5假设）
model = LinearRegression()
model.fit(X_scaled, y)
print("\n【线性回归系数】")
for i, var in enumerate(selected_vars):
    print(f"{var} 对购物行为的影响系数: {model.coef_[i]:.4f}")

# 使用statsmodels获取详细统计量
X_with_const = sm.add_constant(X_scaled)
results = sm.OLS(y, X_with_const).fit()
print("\n【回归模型摘要】")
print(results.summary().tables[1])  # 显示系数表

# 5. 多重共线性检查（VIF）
vif = pd.DataFrame()
vif["变量"] = X_scaled.columns
vif["VIF"] = [variance_inflation_factor(X_scaled.values, i) 
              for i in range(X_scaled.shape[1])]
print("\n【多重共线性检查(VIF)】")
print(vif)

# 6. 方差分析（验证不同信息全面性分组的影响差异）
df['group'] = pd.cut(df['info_completeness'], bins=5, labels=[1, 2, 3, 4, 5])
target_vars = ['website_quality', 'perceived_ease', 'perceived_utility', 'shopping_behavior']

print("\n【方差分析结果】")
for var in target_vars:
    groups = [df[df['group'] == i][var].dropna() for i in range(1, 6) 
              if not df[df['group'] == i][var].empty]
    if len(groups) >= 2:
        f_stat, p_value = f_oneway(*groups)
        print(f"\n{var} 的方差分析:")
        print(f"F统计量: {f_stat:.4f}, p值: {p_value:.4f}, 显著性: {'显著' if p_value < 0.05 else '不显著'}")
        
        # Tukey HSD事后检验
        df_tukey = pd.DataFrame({'value': df[var], 'group': df['group'].astype(str)})
        tukey = pairwise_tukeyhsd(endog=df_tukey['value'], groups=df_tukey['group'], alpha=0.05)
        print("Tukey HSD事后检验:")
        tukey_df = pd.DataFrame(tukey._results_table.data[1:], columns=tukey._results_table.data[0])
        print(tukey_df.head(3))  # 显示前3组对比

# 7. 数据可视化（生成文档中的关键图表）
plt.figure(figsize=(18, 12))

# 图1：感知有用性与信息全面性的线性关系（文档图5延伸）
plt.subplot(2, 2, 1)
sns.regplot(x='info_completeness', y='perceived_utility', data=df, scatter_kws={'alpha':0.5})
plt.title('感知有用性与信息全面性的线性关系')
plt.xlabel('信息全面性得分')
plt.ylabel('感知有用性得分')

# 图2：不同分组的购物行为均值（文档图14延伸）
plt.subplot(2, 2, 2)
group_means = df.groupby('group')['shopping_behavior'].mean()
sns.barplot(x=group_means.index, y=group_means.values, palette='viridis')
plt.title('不同信息全面性分组的购物行为均值')
plt.xlabel('分组编号')
plt.ylabel('购物行为均值')

# 图3：网站质量与感知有用性的关系（文档表12可视化）
plt.subplot(2, 2, 3)
sns.scatterplot(x='website_quality', y='perceived_utility', data=df, alpha=0.6)
plt.title('网站质量对感知有用性的影响')
plt.xlabel('网站质量得分')
plt.ylabel('感知有用性得分')

# 图4：便利性与购物行为的关系（文档H6假设可视化）
plt.subplot(2, 2, 4)
sns.regplot(x='convenience', y='shopping_behavior', data=df)
plt.title('便利性对购物行为的影响')
plt.xlabel('便利性得分')
plt.ylabel('购物行为得分')

plt.tight_layout()
plt.savefig('直播带货消费行为分析.png', dpi=300)
plt.show()

print("\n分析完成，已生成4张关键可视化图表")
