# 4. 线性回归分析（修正绘图部分）
def linear_regression_analysis(df, independent_vars, dependent_var):
    scaler = StandardScaler()
    X = scaler.fit_transform(df[independent_vars])
    y = df[dependent_var].values
    X = sm.add_constant(X)
    
    model = sm.OLS(y, X).fit()
    
    print("\n线性回归分析结果（标准化系数）：")
    print(model.summary2().tables[1])
    
    vif_data = pd.DataFrame()
    vif_data["变量"] = independent_vars
    vif_data["VIF"] = [variance_inflation_factor(X[:, 1:], i) for i in range(X.shape[1]-1)]
    print("\n多重共线性检查（VIF）：")
    print(vif_data[vif_data["VIF"] > 10] if vif_data["VIF"].max() > 10 else "所有变量VIF<10，无严重共线性")
    
    # 修正：将系数转换为Pandas Series再绘图
    coefficients = pd.Series(model.params[1:], index=independent_vars)
    plt.figure(figsize=(12, 6))
    coefficients.plot(kind='bar', color='skyblue')
    plt.title('标准化回归系数可视化')
    plt.ylabel('标准化系数(β)')
    plt.axhline(y=0, color='r', linestyle='-')
    plt.xticks(rotation=90)
    plt.tight_layout()
    plt.show()
    
    return model

# 5. 假设检验
def hypothesis_testing(model, hypotheses):
    print("\n===== 假设检验结果 =====")
    for hyp, (var, expected_direction) in hypotheses.items():
        p_value = model.pvalues[var]
        beta = model.params[var]
        direction_met = (beta > 0 and expected_direction == '正向') or (beta < 0 and expected_direction == '负向')
        significant = "成立" if p_value < 0.05 and direction_met else "不成立"
        print(f"{hyp}: {var} 对因变量有{expected_direction}影响 -> {significant} (p={p_value:.4f}, β={beta:.3f})")
