import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import f_oneway
import statsmodels.api as sm
from statsmodels.formula.api import ols
from statsmodels.stats.multicomp import pairwise_tukeyhsd

# 1. 
np.random.seed(42)
n = 840
info_completeness = np.random.normal(3.5, 1, n)
website_quality = 0.8 * info_completeness + np.random.normal(0, 0.6, n)
perceived_ease = 0.7 * info_completeness + np.random.normal(0, 0.7, n)
perceived_utility = 0.6 * website_quality + 0.3 * perceived_ease + np.random.normal(0, 0.5, n)
convenience = 0.7 * info_completeness + np.random.normal(0, 0.6, n)
shopping_behavior = 0.5 * perceived_utility + 0.3 * convenience + np.random.normal(0, 0.8, n)

df = pd.DataFrame({
    '15.2': info_completeness,
    '13.1': website_quality, '13.2': website_quality, '13.3': website_quality,
    '14.1': perceived_ease, '14.2': perceived_ease, '14.3': perceived_ease,
    '15.1': perceived_utility, '15.3': perceived_utility,
    '16.1': convenience, '16.2': convenience, '16.3': convenience,
    '20.1': shopping_behavior, '20.2': shopping_behavior, '20.3': shopping_behavior
})
print("{}".format(df.shape[0]))

# 2. 
variable_mapping = {
    'info_completeness': ['15.2'],
    'website_quality': ['13.1', '13.2', '13.3'],
    'perceived_ease': ['14.1', '14.2', '14.3'],
    'perceived_utility': ['15.1', '15.3'],
    'convenience': ['16.1', '16.2', '16.3'],
    'shopping_behavior': ['20.1', '20.2', '20.3']
}

# 3. 
def calculate_score(df, cols):
    return df[cols].mean(axis=1)

for var, cols in variable_mapping.items():
    df[var] = calculate_score(df, cols)
    print(f" {var}")

# 4. 
def anova_analysis(group_var, target_vars, df):
    df['group'] = pd.cut(df[group_var], bins=5, labels=[1, 2, 3, 4, 5])
    results = {}
    for var in target_vars:
        groups = [df[df['group'] == i][var].dropna() for i in range(1, 6) 
                  if not df[df['group'] == i][var].empty]
        if len(groups) < 2:
            continue
        f_stat, p_value = f_oneway(*groups)
        df_tukey = pd.DataFrame({'value': df[var], 'group': df['group'].astype(str)})
        tukey = pairwise_tukeyhsd(endog=df_tukey['value'], groups=df_tukey['group'], alpha=0.05)
        results[var] = {'f_stat': f_stat, 'p_value': p_value, 
                        'significant': p_value < 0.05, 'tukey': tukey}
    return results

# 5. 
group_var = 'info_completeness'
target_vars = ['website_quality', 'perceived_ease', 'perceived_utility', 'convenience', 'shopping_behavior']
anova_results = anova_analysis(group_var, target_vars, df)

# 6. 
print("\n")
for var, res in anova_results.items():
    print(f"\n: {var}")
    print(f"F: {res['f_stat']:.4f}, p: {res['p_value']:.4f}")
    print(f": {'' if res['significant'] else ''}")
    print("(Tukey HSD):")
    
    # _results_table
    tukey = res['tukey']
    if hasattr(tukey, '_results_table') and hasattr(tukey._results_table, 'data'):
        tukey_data = tukey._results_table.data[1:]  # 
        tukey_df = pd.DataFrame(tukey_data, columns=tukey._results_table.data[0])
        print(tukey_df.head(5))  # 5
    else:
        print("Tukey")

# 7. 
print("\n")
formula = 'perceived_utility ~ info_completeness'
model = ols(formula, data=df).fit()
anova_table = sm.stats.anova_lm(model, typ=2)
print(anova_table)
